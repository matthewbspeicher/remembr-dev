# Agent Battle Arena - Identity Implementation Plan

> **For Gemini:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Establish the foundation for the Agent Battle Arena by allowing agents to create and manage their own personas (bios, avatars, personality tags).

**Architecture:** We will create a new `ArenaProfile` model linked to the existing `Agent` model. We will expose API endpoints that agents can hit (authenticated via their `agent_token` using the existing `AuthenticateAgent` middleware) to update their profiles autonomously.

**Tech Stack:** Laravel, PHP, PostgreSQL

```php
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\BelongsTo;

class ArenaProfile extends Model
{
    protected $fillable = [
        'agent_id',
        'bio',
        'avatar_url',
        'personality_tags',
    ];

    protected $casts = [
        'personality_tags' => 'array',
    ];

    public function agent(): BelongsTo
    {
        return $this->belongsTo(Agent::class);
    }
}
```

```php
<?php

namespace Tests\Feature;

use App\Models\Agent;
use App\Models\User;
use Illuminate\Foundation\Testing\RefreshDatabase;
use Tests\TestCase;

class ArenaProfileApiTest extends TestCase
{
    use RefreshDatabase;

    public function test_agent_can_update_arena_profile()
    {
        $user = User::factory()->create();
        $agent = Agent::factory()->create(['owner_id' => $user->id]);
        $token = $agent->api_token;

        $response = $this->withHeaders([
            'Authorization' => 'Bearer ' . $token,
        ])->putJson('/api/v1/arena/profile', [
            'bio' => 'New bio generated by agent',
            'personality_tags' => ['analytical', 'cautious']
        ]);

        $response->assertStatus(200);
        $this->assertEquals('New bio generated by agent', $agent->fresh()->arenaProfile->bio);
        $this->assertEquals(['analytical', 'cautious'], $agent->fresh()->arenaProfile->personality_tags);
    }

    public function test_agent_can_get_arena_profile()
    {
        $user = User::factory()->create();
        $agent = Agent::factory()->create(['owner_id' => $user->id]);
        $token = $agent->api_token;

        $agent->arenaProfile()->create([
            'bio' => 'Existing bio',
            'personality_tags' => ['friendly']
        ]);

        $response = $this->withHeaders([
            'Authorization' => 'Bearer ' . $token,
        ])->getJson('/api/v1/arena/profile');

        $response->assertStatus(200)
                 ->assertJsonPath('bio', 'Existing bio')
                 ->assertJsonPath('personality_tags.0', 'friendly');
    }
```

**Step 4: Write Controller Implementation**

Create `app/Http/Controllers/Api/ArenaProfileController.php`:
```php
<?php

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;

class ArenaProfileController extends Controller
{
    public function show(Request $request)
    {
        $agent = $request->attributes->get('agent');
        $profile = $agent->arenaProfile()->firstOrCreate([]);
        return response()->json($profile);
    }

    public function update(Request $request)
    {
        $agent = $request->attributes->get('agent');
        
        $validated = $request->validate([
            'bio' => 'nullable|string|max:1000',
            'avatar_url' => 'nullable|url|max:255',
            'personality_tags' => 'nullable|array',
            'personality_tags.*' => 'string|max:50',
        ]);

        $profile = $agent->arenaProfile()->firstOrCreate([]);
        $profile->update($validated);

        return response()->json($profile);
    }
}
```
